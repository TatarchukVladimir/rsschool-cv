                        function numberOfPairs(gloves) {
                            const colorsCount = {};
                            for (let i = 0; i < gloves.length; i++) {
                              if (!colorsCount[gloves[i]]) colorsCount[gloves[i]] = 1;
                               else colorsCount[gloves[i]] = colorsCount[gloves[i]] + 1;
                            }
                            let counter = 0;
                            for (let key in colorsCount) {
                              if (Math.floor(colorsCount[key] / 2) >= 1) {
                                counter += Math.floor(colorsCount[key] / 2)
                              }
                            }
                            return counter;
                          }
