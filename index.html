function formatDuration (seconds) {
if (seconds === 0) return 'now';
const obj = {};
obj.year = Math.floor(seconds / 365 / 24 / 60 / 60);
obj.day = Math.floor((seconds / 60 / 60 / 24) % 365);
obj.hour = Math.floor((seconds / 60 / 60) % 24);
obj.minute = Math.floor((seconds / 60) % 60);
obj.second = seconds % 60;
let keys = Object.keys(obj);
for ( let i = 0; i < keys.length; i++ ) {
if ((obj[keys[i]]) === 1) {
obj[keys[i]] = `${obj[keys[i]]} ${keys[i]},`;
} else if ((obj[keys[i]]) === 0) {
delete obj[keys[i]];
} else {
obj[keys[i]] = `${obj[keys[i]]} ${keys[i]}s,`
}
}
keys = Object.keys(obj);
for ( let i = 0; i < keys.length; i++ ) {
if (i === (keys.length - 2) ) {
obj[keys[i]] = `${obj[keys[i]].slice(0, -1)} and`;
}
}
let strMassive = Object.values(obj);
let str = strMassive.join(' ');
str = str.slice(0, -1);
return str;
}
